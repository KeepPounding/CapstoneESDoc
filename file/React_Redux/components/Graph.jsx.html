    import React, { Component, PropTypes } from 'react'
import { bindActionCreators } from 'redux'
import FusionCharts from 'fusioncharts'
import charts from 'fusioncharts/fusioncharts.charts'
import reactFusionCharts from 'react-fusioncharts'
import { connect } from 'react-redux'
import * as graphActions from '../actions'
import * as CONSTANTS from '../constants'

charts(FusionCharts)

class Graph extends Component {

	componentWillMount() {
		let self = this;
		const mode		= this.props.mode,
			  sessionId = this.props.sessionId,
			  report    = (this.props.report) ? this.props.report : null,
			  reportId  = (report) ? report.id : null

		if (reportId) {
			if (report && !report.metadata) {
				self.props.actions.retrieveReportMetaData(sessionId, reportId)
			}
		}
	}

	componentDidMount() {
		const graphId     = (this.props.id) ? this.props.id : this.props.graph.Id,
			  graphTypeId = this.props.graph.Graph__c
		let data = (this.props.report && this.props.report.data) ? this.props.report.data : []

		if (graphId && data) {
			this.state = {
				chart : new FusionCharts({
					renderAt   : graphId,
				    type       : graphTypeId,
				    width      : '500',
				    height     : '300',
				    dataFormat : 'json',
				    dataSource : {
				    	"data" : data
				    }
				}).render()
			}
		}
	}

	componentWillReceiveProps(nextProps) {
		let reportId, sessionId, metadata, data
		const mode		= this.props.mode, 
			  report    = (nextProps && nextProps.report) ? nextProps.report : this.props.report,
			  graph     = (nextProps && nextProps.graph) ? nextProps.graph : this.props.graph,
			  chart     = this.state.chart,
			  actions   = this.props.actions,
			  oldGraph  = this.props.graph.Graph__c,
			  newGraph  = nextProps.graph.Graph__c,
			  oldReport = this.props.graph.Report__c,
			  newReport = nextProps.graph.Report__c

		sessionId = this.props.sessionId
		reportId  = (newReport) ? newReport : ((oldReport) ? oldReport : '')

		if (report && !report.metadata && mode == CONSTANTS.READY) {
			metadata = (report.metadata) ? report.metadata : this.props.actions.retrieveReportMetaData(sessionId, reportId)
		} else if (report.metadata) {
			metadata = report.metadata
		}

		if (report && report.metadata && !report.data && mode == CONSTANTS.READY) {
			data = (report.data) ? report.data : this.props.actions.retrieveReportData(sessionId, reportId, report.metadata)
		} else if (report.data) {
			data = report.data
		}

		if (chart && data) {
			const existingData = chart.getJSONData()

			if (existingData.data && existingData.data.length != data.length) {
				chart.setChartData({ data : data })
				chart.render()
			}
		}

		if (oldGraph != newGraph) {
			chart.chartType(newGraph)
			chart.render()
		}
	}

	render() {
		const graphId = (this.props.id) ? this.props.id : this.props.graph.Id

		return (
			<div id={ graphId } />
		);
	}
}

Graph.propTypes = {
	graphs    : PropTypes.array.isRequired,
	reports   : PropTypes.array.isRequired,
	actions   : PropTypes.object.isRequired,
	sessionId : PropTypes.string.isRequired,
	mode      : PropTypes.string.isRequired
};

function mapStateToProps(state) {
  	return {
    	graphs    : state.graphs,
    	reports   : state.reports,
    	mode      : state.mode,
    	sessionId : state.session
	};
}

function mapDispatchToProps(dispatch) {
	return {
		actions : bindActionCreators(graphActions, dispatch)
	}
}

export default connect(mapStateToProps, mapDispatchToProps)(Graph);
