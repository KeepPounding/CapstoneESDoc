    import React, { Component, PropTypes } from 'react'
import { bindActionCreators } from 'redux';
import { connect } from 'react-redux'
import * as CONSTANTS from '../constants'
import * as graphActions from '../actions'
import Dropdown from './Dropdown'
import Error from './Error'
import Header from './Header'
import Button from './Button'
import Modal from './Modal'
import Icon from './Icon'
import Graph from './Graph'
import * as utils from '../libs/utils'

class GraphGuide extends Component {
	render() {
		const mode          = this.props.mode
		const actions       = this.props.actions
		const graph         = this.props.graph
		const sessionId 	= this.props.sessionId
		const cancelEditing = actions.cancelEditing.bind(this)
		const previewId     = 'previewContainer-' + utils.generateGUID()

		let dropdownReports, dropdownGraphTypes,
			reportId, graphTypeId

		var report, buttonText, buttonAction, buttonParams
		
		if (mode == CONSTANTS.EDIT || mode == CONSTANTS.NEW) {
			if (graph && graph.Graph__c && this.props.reports) {
				report = this.props.reports.filter((report) => report.id == graph.Report__c)[0]
			}

			dropdownReports = this.props.reports.map(function(report) {
				return { 
					value : report.id, 
					label : report.name 
				}
			})
			dropdownGraphTypes = this.props.graphTypes.map(function(graphType) {
				return { 
					value : graphType.DeveloperName, 
					label : graphType.MasterLabel
				};
			})

			if (graph.Id != null) {
				buttonText   = CONSTANTS.MODAL_BUTTON_UPDATE
				buttonAction = actions.updateGraph.bind(this, sessionId, graph.Id, graph.Report__c, graph.Graph__c)
			} else {
				buttonText   = CONSTANTS.MODAL_BUTTON_CREATE
				buttonAction = actions.createGraph.bind(this, sessionId, graph.Report__c, graph.Graph__c)
				buttonParams = []
			}

			return (
				<Modal>
					<div className="build">
						<Header label={ CONSTANTS.MODAL_HEADER } />

						<Dropdown label={ CONSTANTS.CHOOSE_REPORT } data={ dropdownReports } preSelectedValue={ graph.Report__c } onChange={ actions.updateReportId } />

						<Error>{ CONSTANTS.ERROR_MESSAGE }</Error>

						<Dropdown label={ CONSTANTS.CHOOSE_GRAPH } data={ dropdownGraphTypes } preSelectedValue={ graph.Graph__c } onChange={ actions.updateGraphId } />

						<div className="parameters">
							<label>
								{ CONSTANTS.CHECKBOX_GROUP_HEADER }
							</label>

							<ul style={{ listStyleType : 'none' }}>
								<li>
									<input id="1" type="checkbox" className="checkbox" value="0" />
									<label htmlFor="1">{ CONSTANTS.CHECKBOX_1 }</label>
								</li>

								<li>
									<input id="2" type="checkbox" className="checkbox" value="1" />
									<label htmlFor="2">{ CONSTANTS.CHECKBOX_2 }</label>
								</li>

								<li>
									<input id="3" type="checkbox" className="checkbox" value="2" />
									<label htmlFor="3">{ CONSTANTS.CHECKBOX_3 }</label>
								</li>

								<li>
									<input id="4" type="checkbox" className="checkbox" value="3" />
									<label htmlFor="4">{ CONSTANTS.CHECKBOX_4 }</label>
								</li>
							</ul>
						</div>
					</div>

					<div className="preview">
						<Icon name="close" onClick={ cancelEditing } />

						<h1>{ CONSTANTS.PREVIEW_HEADER }</h1>

						<div id="chartPreview">
							<Graph key= { previewId } id={ previewId } graph={ graph } report={ report } />
						</div>

						<Button label={ buttonText } onClick={ buttonAction } params={ buttonParams } />
					</div>
					<div style={{ clear : 'both' }} />
				</Modal>
			);
		} else {
			return (
				<div />
			);
		}
	}
}

GraphGuide.propTypes = {
	actions    : PropTypes.object.isRequired,
	graphTypes : PropTypes.array.isRequired,
	reports    : PropTypes.array.isRequired,
	sessionId  : PropTypes.string.isRequired,
	mode       : PropTypes.string.isRequired
};

function mapStateToProps(state) {
  	return {
  		sessionId  : state.session,
  		graphTypes : state.graphTypes,
  		reports    : state.reports,
  		graph 	   : state.selectedGraph,
  		mode       : state.mode
	};
}

function mapDispatchToProps(dispatch) {
	return {
		actions : bindActionCreators(graphActions, dispatch)
	}
}

export default connect(mapStateToProps, mapDispatchToProps)(GraphGuide);
