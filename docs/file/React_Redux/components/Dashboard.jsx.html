    import React, { Component, PropTypes } from 'react'
import { bindActionCreators } from 'redux';
import { connect } from 'react-redux'
import * as graphActions from '../actions'
import Graph from './Graph'
import GraphGuide from './GraphGuide'
import Icon from './Icon'
import Alert from 'react-s-alert'

import 'react-s-alert/dist/s-alert-default.css'
import 'react-s-alert/dist/s-alert-css-effects/scale.css'

export const buildGraphs = (props, actions, graphs, reports) => {
	return graphs.map(function(graph) {
		const report 	   = reports.filter((report) => report.id == graph.Report__c)[0],
			  editGraph    = actions.editGraph.bind(this, graph),
			  refreshGraph = actions.refreshGraph.bind(this, props.sessionId, graph.Id),
			  deleteGraph  = actions.deleteGraph.bind(this, props.sessionId, graph.Id);

		return (
			<li key={ graph.Id }>
				<div className="toolbar">
					<Icon name="pencil" onClick={ editGraph } />
					<Icon name="refresh" onClick={ refreshGraph } />
					<Icon name="trash" onClick={ deleteGraph } />
				</div>

				<Graph key={ graph.Id } id={ graph.Id } graph={ graph } report={ report } />
			</li>
		)
	})
}

class Dashboard extends Component {
	componentWillMount() {
		const sessionId = this.props.sessionId;

		this.props.actions.retrieveReports(sessionId);
		this.props.actions.retrieveGraphs(sessionId);
		this.props.actions.retrieveGraphTypes(sessionId);
	}

	render() {
		const actions  = this.props.actions
		const graphs   = buildGraphs(this.props, actions, this.props.graphs, this.props.reports)
		const newGraph = actions.newGraph.bind(this);

		return (
			<div id="charting">
				<ul className="graphs">{ graphs }</ul>
				<ul className="addNewGraph">
					<li>
						<Icon id="addNewChart" name="plus" onClick={ newGraph } />
					</li>
				</ul>

				<GraphGuide />

				<Alert stack={{limit: 7}} />
			</div>
		);
	}
}

Dashboard.propTypes = {
	graphs     : PropTypes.array.isRequired,
	reports    : PropTypes.array.isRequired,
	actions    : PropTypes.object.isRequired,
	sessionId  : PropTypes.string.isRequired
};

function mapStateToProps(state) {
  	return {
    	graphs     : state.graphs,
    	reports    : state.reports,
    	sessionId  : state.session
	};
}

function mapDispatchToProps(dispatch) {
	return {
		actions : bindActionCreators(graphActions, dispatch)
	}
}

export default connect(mapStateToProps, mapDispatchToProps)(Dashboard);
